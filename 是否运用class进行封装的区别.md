```cpp
// 引入标准库：就像是在造机器人前，先准备好基础工具箱
#include <iostream> // 允许我们使用 std::cout 在屏幕上打印输出信息
#include <string>   // 允许我们使用 std::string 来处理文本字符串（比如机器人的名字）

// 定义一个名叫 MobileRobot 的“类”（也就是机器人的设计图纸）
class MobileRobot {

// private（私有区域）：这里的变量是机器人的“内部核心零件”。
// 意思是：除了机器人自己（也就是类内部的函数），外部代码（比如 main 函数）绝对不能直接修改这些数据，这叫“封装”。
private: 
    std::string robot_name;   // 机器人的名字
    double battery_level;     // 机器人的当前电量
    double current_velocity;  // 机器人的当前速度

// public（公共区域）：这里的函数是机器人暴露给外部的“遥控器按键”。
// 外部代码可以通过调用这些函数，来安全地指挥机器人或查看状态。
public: 
    
    // 【构造函数】：名字必须和类名一模一样，且没有返回值。
    // 它的作用是“出厂设置”——当你用这个图纸造一台新机器人时，这个函数会自动运行。
    // "(std::string name)" 表示造机器人时，必须给它起个名字。
    // ":" 后面的部分叫“初始化列表”，是在给内部变量赋初始值。
    MobileRobot(std::string name) : 
        robot_name(name),       // 把外面传进来的名字赋给私有变量 robot_name
        battery_level(100.0),   // 新机器出厂，电量默认充满 100.0%
        current_velocity(0.0)   // 刚出厂还没动，速度默认为 0.0 m/s
    {} // 大括号里可以写其他出厂需要做的操作，这里为空

    // 【成员函数】：机器人拥有的具体技能/动作。
    // 这个函数用来设置机器人的速度。因为它是 public 的，所以可以用遥控器（外部代码）按下它。
    void setVelocity(double v) {
        
        // 在执行动作前，机器人会先检查自己的私有状态（电量是否大于 5.0）
        if (battery_level > 5.0) {
            current_velocity = v;  // 改变自身的速度状态
            battery_level -= 0.5;  // 每次设置速度，消耗 0.5 的电量（修改了自身的电量状态）
            
            // 打印当前的最新状态
            std::cout << robot_name << " 速度已设为 " << current_velocity 
                      << " m/s. 剩余电量: " << battery_level << "%\n";
        } else {
            // 如果电量不够，拒绝执行，保护内部电池
            std::cout << robot_name << " 电量不足，无法移动！\n";
        }
    }
}; // 注意：定义完一个类，最后的大括号外面必须有一个分号 ;

// main 函数是整个程序的入口，程序从这里开始一行行往下跑
int main() {
    
    // 根据 MobileRobot 这个“图纸”，真正制造出两台物理上独立的“实体机器人”（对象）
    // 制造 A 的时候，给它命名为 "TurtleBot_1"，此时会自动调用上面的【构造函数】
    MobileRobot robot_A("TurtleBot_1"); 
    // 制造 B 的时候，给它命名为 "TurtleBot_2"
    MobileRobot robot_B("TurtleBot_2"); 

    // 调用对象的公共函数（按下遥控器）
    // 虽然调用的都是 setVelocity，但因为是针对不同对象调用的，所以它们互不干扰！
    
    // 操作 A：robot_A 内部的 velocity 变成了 1.2，电量变成了 99.5
    robot_A.setVelocity(1.2); 
    
    // 操作 B：robot_B 内部的 velocity 变成了 0.8，电量变成了 99.5
    // 注意：B 的操作完全不会影响到 A 的电量和速度！
    robot_B.setVelocity(0.8); 

    return 0; // 程序顺利结束
}
```


这是一个非常好的问题！把同一个业务场景用两种不同的方式写出来，最能体现它们的核心差异。

我们现在就把**例子一（控制机器人速度）**的逻辑，从“类（面向对象）”改写成“普通函数（面向过程）”的方式。

### 改写：用“普通函数”实现例子一

在使用普通函数时，我们必须把**“数据”**和**“动作”**拆分开。我们会用一个 `struct`（结构体）只装数据，然后用一个独立的全局函数来操作这些数据。

太棒了！你现在拿出的这段代码，展示的是另一种编程思想——**“面向过程（Procedural Programming）”**，或者说**“C语言风格的结构体用法”**。

如果说上一段代码（面向对象）是把“数据（电量）”和“动作（设置速度）”绑定在一起做成了一个**智能机器人**；那么这段代码就是把“数据”和“动作”彻底分开：数据是一个**没有生命的零件箱**，动作是一个**外部的加工厂**。

为了帮你更好地建立直观认知，你可以先看看下面这种思想对比图：



下面我同样为你加上了超级详细的新手批注，重点帮你理解这两者的区别以及 C++ 中的核心概念：

### 1. 带有详细批注的代码

```cpp
// 引入标准工具箱
#include <iostream>
#include <string>

// 1. 数据部分：定义一个名叫 MobileRobotData 的“结构体（struct）”
// 在这里，它就是一个纯粹的“纸箱子”，里面只装数据（零件），没有任何动作。
// 注意：在 struct 中，如果没有特别标明 private，里面的东西默认全都是 public（公开的）！
struct MobileRobotData {
    std::string robot_name;   // 机器人的名字
    double battery_level;     // 电量
    double current_velocity;  // 速度
};

// 2. 动作部分：这是一个【全局普通函数】（也就是外部的加工厂）
// 它不属于任何机器人，它独立存在。
// 核心知识点【& 符号】：MobileRobotData& robot 这里的 '&' 叫做“引用”。
// 意思是：请把真正的那个机器人实体送进工厂，而不是送一个复制品过来。
// 这样工厂里对 robot.battery_level 的修改，才能真正作用到外面的机器人身上。
void setRobotVelocity(MobileRobotData& robot, double v) {
    
    // 因为函数在外部，它不知道自己要操作谁，所以必须加上前缀 "robot." 
    // 意思是：“查看被送进来的这台 robot 的电量”
    if (robot.battery_level > 5.0) {
        robot.current_velocity = v; // 修改这台机器人的速度
        robot.battery_level -= 0.5; // 消耗这台机器人的电量
        
        std::cout << robot.robot_name << " 速度已设为 " << robot.current_velocity 
                  << " m/s. 剩余电量: " << robot.battery_level << "%\n";
    } else {
        std::cout << robot.robot_name << " 电量不足，无法移动！\n";
    }
}

int main() {
    // 制造两个装满初始数据的“纸箱子”（数据实体）
    // 大括号 {} 是一种快速初始化的语法，按顺序给 name, battery, velocity 赋值
    MobileRobotData robot_A = {"TurtleBot_1", 100.0, 0.0};
    MobileRobotData robot_B = {"TurtleBot_2", 100.0, 0.0};

    // 把机器人 A 送进外部的加工厂，并下达目标速度 1.2 的指令
    setRobotVelocity(robot_A, 1.2); 
    // 把机器人 B 送进外部的加工厂，下达目标速度 0.8 的指令
    setRobotVelocity(robot_B, 0.8);

    // 【危险操作示范】：破坏规则
    // 因为 struct 里的数据默认是完全公开的，没有 private 的保护伞
    // 外部代码根本不需要经过加工厂的检查（不需要判断电量是否大于5），就可以直接把手伸进箱子里乱改！
    robot_A.battery_level = 9999.0; // 逻辑漏洞：物理学不存在了，电量被非法篡改成一万！

    return 0;
}
```

---

### 2. 核心知识点拆解（进阶对比）

*   **`struct` (结构体) vs `class` (类)**
    *   在 C++ 中，`struct` 和 `class` 其实非常像。但它们最大的区别在于**默认权限**：`struct` 默认是“敞着门”的（`public`），谁都能进来看和拿；`class` 默认是“锁着门”的（`private`），必须通过主人允许（公共函数）才能操作。
    *   这段代码展示的就是没有“门锁”的后果：数据暴露在外，极其容易在大型项目中被其他程序员不小心改错（比如那句危险的 `9999.0`）。这就是为什么上一段代码的 **“封装（Encapsulation）”** 如此重要。
*   **什么是“引用传递 (`&`)”？**
    *   在 `void setRobotVelocity(MobileRobotData& robot, double v)` 中，那个 `&` 极其关键！
    *   如果**不加 `&`**（写成 `MobileRobotData robot`），这叫“值传递”。C++ 会偷偷**复印**一份 `robot_A` 的数据交给工厂。工厂给复印件减了 0.5 的电量，但真正的 `robot_A` 电量依然是 100。
    *   **加上 `&`**（引用传递），就相当于牵了一根红线，让函数里的 `robot` 直接绑定到外面的 `robot_A` 身上，修改的就是本体。
*   **动作与数据的分离（面向过程的痛点）**
    *   在上一段代码中，设置速度是写成 `robot_A.setVelocity(1.2);`，读起来像人类语言：“机器人A，请设置你的速度”。
    *   在这段代码中，写成了 `setRobotVelocity(robot_A, 1.2);`，读起来像：“设置速度函数，请处理机器人A”。
    *   当程序越来越大，有几十种机器人、几百个函数时，把函数单独放在外面满天飞，管理起来会极其痛苦。所以现代大型 C++ 开发（尤其是机器人开发领域，比如 ROS），都强烈倾向于使用上一段代码那种**面向对象（OOP）**的写法。

---

### 核心对比分析

现在我们把原版（写在 Class 里的成员函数）和这版（直接写的普通函数）放在一起比一比：

#### 1. 语法的视觉体验（谁是主语？）
*   **Class 写法：** `robot_A.setVelocity(1.2);`
    *   *解读：* **“机器人 A，请你把速度设为 1.2。”** 主语明确，动作是由对象自己发出的，非常符合人类直觉。
*   **普通函数写法：** `setRobotVelocity(robot_A, 1.2);`
    *   *解读：* **“系统，请你把机器人 A 的速度设为 1.2。”** 动作是由第三方（上帝视角）执行的，机器人只是一块被操作的“肉”。

#### 2. 数据的安全性（封装性）
这是这两者在软件工程中最大的区别！
*   **Class 写法（赢）：** 电量 `battery_level` 是 `private`（私有的）。除了 `setVelocity` 这个内部函数，**没有任何人能在外面直接修改电量**。这就防止了别人在代码里写出 `robot_A.battery_level = 9999` 这种违背物理常识的 bug。
*   **普通函数写法（输）：** 为了让外部的普通函数能修改电量，`MobileRobotData` 里的变量必须是公开的（`public`）。这意味着代码里的任何人、任何地方都可以绕过检查，直接修改 `robot_A.battery_level`。当项目变大时，数据被谁改乱了根本查不到。

#### 3. 隐藏参数 (`this` 指针)
*   **普通函数写法：** 你可以看到，我们在参数里明确写了 `(MobileRobotData& robot, double v)`，明确告诉函数要操作哪个机器人。
*   **Class 写法：** 代码里只写了 `(double v)`。为什么它知道要改哪个机器人的电量？因为 C++ 编译器在底层偷偷给你塞了一个名叫 `this` 的指针。当你调用 `robot_A.setVelocity(1.2)` 时，编译器实际上把它转换成了类似 `setVelocity(&robot_A, 1.2)` 的形式。**写在 Class 里面，就是一种优雅的“语法糖”**，省去了你每次都要把对象传进去的麻烦。

### 总结

在这个**特定的场景**（控制实体机器人的状态）下：

*   使用 **Class 成员函数** 是**绝对正确**的做法（面向对象编程）。因为它能把机器人的状态紧紧锁死（封装）在内部，对外只提供安全的操作接口。这正是 ROS 2 中大量使用 Class（比如继承 `rclcpp::Node`）的原因。
*   如果使用 **普通函数**，虽然逻辑上跑得通，但代码结构松散，数据极其不安全，属于不良的代码设计。

**记住一个原则：**
如果动作会**改变某种特定实体的状态**（改属性、改状态），用 **Class 函数**。
如果动作只是**算个数、转个格式**（交并比、字符串拼接），用 **普通函数**。
